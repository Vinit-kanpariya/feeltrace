<interfaces>
API contracts (from Plans 05 and 06):

POST /api/analyze
  Request: { url: string }
  Success: 202 { jobId: string }
  Error (SSRF): 422 { error: string, code: 'BLOCKED_IP' | 'BLOCKED_SCHEME' | 'INVALID_URL' | 'DNS_RESOLUTION_FAILED' }
  Error (queue full): 503 plain text "Service busy. Please try again shortly."
  Error (rate limit): 429 plain text "Too many requests. Try again in {X} minutes." + Retry-After header

GET /api/jobs/[jobId]
  Success: 200 { status: 'pending'|'crawling'|'extracting'|'analyzing'|'complete'|'failed', error_message?: string }
  Error: 404 { error: 'Job not found' }

GET /api/results/[jobId]
  Success (complete): 200 { id, jobId, narrative, issues, edges, created_at }
  Error: 404 { error: '...' }

From src/types/job.ts:
  type JobStatus = 'pending' | 'crawling' | 'extracting' | 'analyzing' | 'complete' | 'failed'
  interface JobStatusResponse { status: JobStatus; error_message?: string }
  interface AnalyzeResponse { jobId: string }
  interface AnalyzeErrorResponse { error: string; code?: string }
</interfaces>
